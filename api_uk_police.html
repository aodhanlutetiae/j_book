# send query via variables and f-string

endpoint = 'https://data.police.uk/api'
method = 'stops-street'
lat = "52.629729"
long = "-1.131592"
month = "2021-04"

query = f'{endpoint}/{method}?lat={lat}&lng={long}&date={month}'
print(query)

req = requests.get(query)
req
